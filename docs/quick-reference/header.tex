\usepackage{graphicx}
\usepackage{fancyhdr}
\usepackage{ifthen}
\usepackage{etoolbox}
\pagestyle{fancy}
\fancyhf{}
\setlength{\headsep}{0pt}
% ---- header image -------------------------------------------------------
\newcommand{\headerimage}{../../images/header.png}
\ifthenelse{\equal{\headerimage}{}}{}{%
  \newlength{\hdrheight}%
  \newsavebox{\hdrimgbox}%
  \AtBeginDocument{%
    \savebox{\hdrimgbox}{\hspace*{-\dimexpr(\paperwidth-\textwidth)/2\relax}\includegraphics[width=\paperwidth,keepaspectratio]{\headerimage}}%
    \settoheight{\hdrheight}{\usebox{\hdrimgbox}}%
  }%
  \fancyhead[L]{\raisebox{\dimexpr 1in+\voffset+\topmargin+\headheight-\hdrheight\relax}[0pt][0pt]{\usebox{\hdrimgbox}}}%
}
% ---- footer image -------------------------------------------------------
\newcommand{\footerimage}{../../images/footer.png}
\ifthenelse{\equal{\footerimage}{}}{}{%
  \newsavebox{\ftrimgbox}%
  \AtBeginDocument{%
    \savebox{\ftrimgbox}{\hspace*{-\dimexpr(\paperwidth-\textwidth)/2\relax}\includegraphics[width=\paperwidth,keepaspectratio]{\footerimage}}%
  }%
  \fancyfoot[L]{\raisebox{\dimexpr-\paperheight+1in+\voffset+\topmargin+\headheight+\headsep+\textheight+\footskip\relax}[0pt][0pt]{\usebox{\ftrimgbox}}}%
}
% ---- page numbers -------------------------------------------------------
\newcommand{\pagenumbers}{true}
\ifthenelse{\equal{\pagenumbers}{true}}{%
  \fancyfoot[R]{\thepage}%
}{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
% ---- title page: header/footer images but no page number -------------------
\fancypagestyle{titlepagestyle}{%
  \fancyhf{}
  \ifthenelse{\equal{\headerimage}{}}{}{%
    \fancyhead[L]{\raisebox{\dimexpr 1in+\voffset+\topmargin+\headheight-\hdrheight\relax}[0pt][0pt]{\usebox{\hdrimgbox}}}%
  }
  \ifthenelse{\equal{\footerimage}{}}{}{%
    \fancyfoot[L]{\raisebox{\dimexpr-\paperheight+1in+\voffset+\topmargin+\headheight+\headsep+\textheight+\footskip\relax}[0pt][0pt]{\usebox{\ftrimgbox}}}%
  }
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}
\AtBeginDocument{\apptocmd{\maketitle}{\thispagestyle{titlepagestyle}}{}{}}
% Apply the same style to the first page of each chapter
\fancypagestyle{plain}{%
  \fancyhf{}
  \ifthenelse{\equal{\headerimage}{}}{}{%
    \fancyhead[L]{\raisebox{\dimexpr 1in+\voffset+\topmargin+\headheight-\hdrheight\relax}[0pt][0pt]{\usebox{\hdrimgbox}}}%
  }
  \ifthenelse{\equal{\footerimage}{}}{}{%
    \fancyfoot[L]{\raisebox{\dimexpr-\paperheight+1in+\voffset+\topmargin+\headheight+\headsep+\textheight+\footskip\relax}[0pt][0pt]{\usebox{\ftrimgbox}}}%
  }
  \ifthenelse{\equal{\pagenumbers}{true}}{%
    \fancyfoot[R]{\thepage}%
  }{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}
% ---- list of figures in table of contents ----------------------------------
\pretocmd{\listoffigures}{\addcontentsline{toc}{section}{\listfigurename}}{}{}
% ---- new page for every chapter -----------------------------------------
\newcommand{\chapterbreak}{true}
\ifthenelse{\equal{\chapterbreak}{true}}{%
  \pretocmd{\section}{\clearpage}{}{}%
}{}
