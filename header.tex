\usepackage{graphicx}
\usepackage{fancyhdr}
\usepackage{ifthen}
\usepackage{etoolbox}
\pagestyle{fancy}
\fancyhf{}
\setlength{\headsep}{0pt}
\newlength{\hdrheight}
% ---- header image -------------------------------------------------------
\newcommand{\headerimage}{images/header.png}
\ifthenelse{\equal{\headerimage}{}}{}{%
  \AtBeginDocument{\settoheight{\hdrheight}{\includegraphics[width=\paperwidth,keepaspectratio]{\headerimage}}}%
  \fancyhead[L]{\raisebox{\dimexpr 1in+\voffset+\topmargin+\headheight-\hdrheight\relax}[0pt][0pt]{\hspace*{-\dimexpr(\paperwidth-\textwidth)/2\relax}\includegraphics[width=\paperwidth,keepaspectratio]{\headerimage}}}%
}
% ---- footer image -------------------------------------------------------
\newcommand{\footerimage}{images/footer.png}
\ifthenelse{\equal{\footerimage}{}}{}{%
  \fancyfoot[L]{\raisebox{\dimexpr-\paperheight+1in+\voffset+\topmargin+\headheight+\headsep+\textheight+\footskip\relax}[0pt][0pt]{\hspace*{-\dimexpr(\paperwidth-\textwidth)/2\relax}\includegraphics[width=\paperwidth,keepaspectratio]{\footerimage}}}%
}
% ---- page numbers -------------------------------------------------------
\newcommand{\pagenumbers}{true}
\ifthenelse{\equal{\pagenumbers}{true}}{%
  \fancyfoot[R]{\thepage}%
}{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
% Apply the same style to the first page of each chapter
\fancypagestyle{plain}{%
  \fancyhf{}
  \ifthenelse{\equal{\headerimage}{}}{}{%
    \fancyhead[L]{\raisebox{\dimexpr 1in+\voffset+\topmargin+\headheight-\hdrheight\relax}[0pt][0pt]{\hspace*{-\dimexpr(\paperwidth-\textwidth)/2\relax}\includegraphics[width=\paperwidth,keepaspectratio]{\headerimage}}}%
  }
  \ifthenelse{\equal{\footerimage}{}}{}{%
    \fancyfoot[L]{\raisebox{\dimexpr-\paperheight+1in+\voffset+\topmargin+\headheight+\headsep+\textheight+\footskip\relax}[0pt][0pt]{\hspace*{-\dimexpr(\paperwidth-\textwidth)/2\relax}\includegraphics[width=\paperwidth,keepaspectratio]{\footerimage}}}%
  }
  \ifthenelse{\equal{\pagenumbers}{true}}{%
    \fancyfoot[R]{\thepage}%
  }{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}
% ---- new page for every chapter -----------------------------------------
\newcommand{\chapterbreak}{true}
\ifthenelse{\equal{\chapterbreak}{true}}{%
  \pretocmd{\section}{\clearpage}{}{}%
}{}
